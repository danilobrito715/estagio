\section*{Plano de Aula: Lançamento Vertical}

\textbf{Objetivo:} Apresentar o movimento vertical e discutir gravidade com apoio do Arduino.

\textbf{Materiais:}
\begin{itemize}
  \item Arduino Uno
  \item LED
  \item Resistores
  \item Cabos de conexão
\end{itemize}

\textbf{Atividades:}
\begin{itemize}
  \item Montar o circuito básico.
  \item Executar o código de contagem regressiva.
  \item Discutir o conceito de tempo e movimento.
\end{itemize}
